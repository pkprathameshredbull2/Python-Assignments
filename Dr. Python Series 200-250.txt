206)  How to Slice Dictionary in Python ?

def dict_slice() :
    my_details = {
        "first_name" : "Prathamesh",
        "middle_name" : "Vijay" ,
        "last_name" : "Kadam" ,
        "Address" : "A/806 Cerez Malpani HSG SOC ,near G K Arcos , Gaikwad Nagar ,Punawale, Pune 411088",
        "Age" : 34 ,
        "Mobile_Number" : 8097748413
    }

    key_slice = ["first_name",
                 "middle_name",
                 "last_name"]

    sliced_dict = { key : my_details[key] for key in key_slice}
    print(sliced_dict)
	
# How <sliced_dict = { key : my_details[key] for key in key_slice} > works ? , it is as below
    for key in key_slice :
        print(f"{key} : {my_details[key]}")
